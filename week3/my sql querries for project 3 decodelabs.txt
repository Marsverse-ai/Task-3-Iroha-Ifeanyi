my sql querries for project 3 decodelabs



SELECT
COUNT(coupon_code) as no_of_missing_coupon_codes
FROM ecommerce_orders
where coupon_code = 'N/A';


SELECT COUNT(*) AS total_records FROM ecommerce_orders;

SELECT SUM(total_price) AS total_revenue FROM ecommerce_orders;

SELECT ROUND(AVG(total_price), 2) AS average_revenue FROM ecommerce_orders;

SELECT COUNT(quantity) AS total_quantity_sold FROM ecommerce_orders;

SELECT ROUND(AVG(quantity),2) AS average_quantity_sold FROM ecommerce_orders;

SELECT MAX(total_price) AS largest_sale FROM ecommerce_orders;

SELECT MIN(total_price) AS smallest_sale FROM ecommerce_orders;



SELECT 
	payment_method,
	COUNT(quantity) as total_transactions
FROM ecommerce_orders
GROUP BY payment_method
ORDER BY total_transactions;



SELECT
 product,
COUNT(quantity) AS total_orders
FROM ecommerce_orders
GROUP BY 1
ORDER BY 2 DESC;



SELECT
	referral_source,
	COUNT(quantity) as total_customer_accquisition
FROM ecommerce_orders
GROUP BY 1
ORDER BY 2 DESC;



SELECT
	order_status,
	COUNT(quantity) as order_status_count
FROM ecommerce_orders
GROUP BY 1
ORDER BY 2 DESC;
